    <script>
        let customQuestions = JSON.parse(localStorage.getItem('custom_exam_questions')) || [];

        function updatePreview() {
            const list = document.getElementById("questions-preview-list");
            document.getElementById("q-count").innerText = customQuestions.length;
            
            if (customQuestions.length === 0) {
                list.innerHTML = `<p class="text-sm text-slate-500 italic">अजून एकही प्रश्न जोडलेला नाही. डीफॉल्ट प्रश्न वापरले जातील.</p>`;
                return;
            }

            list.innerHTML = "";
            customQuestions.forEach((item, idx) => {
                const div = document.createElement("div");
                div.className = "bg-slate-900 p-3 rounded-lg border border-slate-700 text-sm flex justify-between items-start";
                div.innerHTML = `
                    <div>
                        <span class="text-blue-400 font-bold">प्रश्न ${idx+1}:</span> ${item.q}
                        <div class="text-xs text-emerald-400 mt-1">बरोबर पर्याय: ${String.fromCharCode(65 + item.a)}</div>
                    </div>
                    <button onclick="deleteQuestion(${idx})" class="text-rose-500 hover:text-rose-400 text-xs font-medium ml-4 mt-0.5">हटवा</button>
                `;
                list.appendChild(div);
            });
        }

        function addQuestionToPaper(event) {
            event.preventDefault();
            const newQ = {
                q: document.getElementById("input-q").value,
                o: [
                    document.getElementById("input-o0").value,
                    document.getElementById("input-o1").value,
                    document.getElementById("input-o2").value,
                    document.getElementById("input-o3").value
                ],
                a: parseInt(document.getElementById("input-correct").value)
            };
            customQuestions.push(newQ);
            localStorage.setItem('custom_exam_questions', JSON.stringify(customQuestions));
            document.getElementById("question-setter-form").reset();
            updatePreview();
        }

        function deleteQuestion(idx) {
            customQuestions.splice(idx, 1);
            localStorage.setItem('custom_exam_questions', JSON.stringify(customQuestions));
            updatePreview();
        }

        function clearAllQuestions() {
            if(confirm("सर्व प्रश्न हटवायचे आहेत का?")) {
                customQuestions = [];
                localStorage.removeItem('custom_exam_questions');
                updatePreview();
            }
        }

        window.onload = updatePreview;
    </script>
